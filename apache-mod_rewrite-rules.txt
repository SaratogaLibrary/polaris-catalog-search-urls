RewriteRule ^cat/ti/(.+)$			http://catalog.mylibrary.org/polaris/view.aspx?title=$1 [L,R=301]

RewriteRule ^cat/au/(.+)$			http://catalog.mylibrary.org/polaris/view.aspx?author=$1 [L,R=301]

RewriteRule ^cat/(.+)$				http://catalog.mylibrary.org/polaris/view.aspx?keyword=$1 [L,R=301]

</IfModule>
